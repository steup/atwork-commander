413  QString printArenaStateList(std::vector<atwork_commander_msgs::Workstation> ws_list, bool print_empty_container = true)
414  {
415  using namespace atwork_commander_msgs;
416 
417  QString arenaString("");
418  const QString indent(" ");
419  //workstation (ws), container (c; EMPTY, blue, red, tiles), object (o)
420  std::map<std::string, std::map<uint16_t, std::vector<uint16_t>>> target_map;
421 
422  std::function<void(int, std::string)> add_line = [&arenaString, &indent](int indents, std::string text) {
423  for (int i = 0; i < indents; i++) {
424  arenaString.append(indent);
425  }
426  arenaString.append(QString::fromStdString(text));
427  arenaString.append("\n");
428  };
429 
430  auto is_object = [](uint16_t object) -> bool { return ((object >= Object::ATWORK_START && object < Object::ATWORK_END) || (object >= Object::ROCKIN_START && object < Object::ROCKIN_END)); };
431  auto is_container = [](uint16_t container) -> bool { return (container >= Object::CONTAINER_START && container < Object::CONTAINER_END); };
432 
433  for (auto& ws : ws_list) {
434  for (auto& o : ws.objects) {
435  if (is_object(o.object)) {
436  target_map[ws.workstation_name][o.target].push_back(o.object);
437  }
438  if (is_container(o.object) && print_empty_container) {
439  target_map[ws.workstation_name][o.target].push_back(o.object);
440  }
441  }
442  }
443 
444  for (auto ws_it = target_map.begin(); ws_it != target_map.end(); ws_it++) {
445  if (ws_it->second.size() == 0) {
446  continue;
447  }
448  add_line(0, "[" + ws_it->first + "]");
449 
450  auto ws = ws_it->second;
451  if (ws.find(Object::EMPTY) != ws.end()) {
452  auto c = ws[Object::EMPTY];
453  for (auto& o : c) {
454  if (is_container(o)) { //special for print_empty_container
455  if (ws.find(o) == ws.end()) {
456  add_line(1, std::string("[") + atwork_commander_msgs::objectName(o) + "]");
457  }
458  continue;
459  }
460  add_line(1, atwork_commander_msgs::objectName(o));
461  }
462  }
463 
464  for (auto c_it = ws.begin(); c_it != ws.end(); c_it++) {
465  if (c_it->first == Object::EMPTY) {
466  continue;
467  }
468  add_line(1, std::string("[") + atwork_commander_msgs::objectName(c_it->first) + "]");
469  for (auto& o : c_it->second) {
470  add_line(2, atwork_commander_msgs::objectName(o));
471  }
472  }
473  }
474 
475  return arenaString;
476  }
