106  void stateUpdate() {
107  switch( m_state.state ) {
108  case ( State::IDLE ):
109  if ( !m_state.robots.empty() && m_checked ) {
110  ROS_DEBUG_STREAM_NAMED("state_tracker", "[REFBOX] Robots registered and task generated! I am READY :-)");
111  m_state.state = State::READY;
112  }
113  break;
114  case ( State::READY ):
115  if ( m_state.robots.empty() || ! m_checked ) {
116  ROS_DEBUG_STREAM_NAMED("state_tracker", "[REFBOX] Lost all robots or generated task! Going back to IDLE!");
117  m_state.state = State::IDLE;
118  break;
119  }
120  if ( m_state.end > ros::Time::now() ) {
121  ROS_DEBUG_STREAM_NAMED("state_tracker", "[REFBOX] Received start command going to PREPARATION ending at " << m_state.end );
122  m_state.state = State::PREPARATION;
123  m_task_timer = m_nh.createTimer( m_state.task.prep_time, &StateTracker::taskUpdate, this, true );
124  }
125  break;
126  case ( State::PREPARATION ):
127  if ( m_state.end < ros::Time::now() ) {
128  ROS_DEBUG_STREAM_NAMED("state_tracker", "[REFBOX] Prep Time over going to EXECUTION ending at " << m_state.end );
129  m_state.state = State::EXECUTION;
130  m_state.end += m_state.task.exec_time;
131  m_task_timer = m_nh.createTimer( m_state.task.exec_time, &StateTracker::taskUpdate, this, true );
132  }
133  break;
134  case ( State::EXECUTION ) :
135  if ( m_state.end < ros::Time::now() ) {
136  ROS_DEBUG_STREAM_NAMED("state_tracker", "[REFBOX] Task Execution time over! Task finished! Going back to READY");
137  m_state.state = State::READY;
138  }
139  break;
140  case ( State::FAILURE ) : ROS_ERROR_STREAM_NAMED("state_tracker",
141  "[REFBOX] In failed state, will not accept any commands! Please restart!");
142  break;
143  default : ROS_ERROR_STREAM_NAMED("state_tracker",
144  "[REFBOX] Invalid state encountered " << m_state.state << "!");
145  }
146  ROS_DEBUG_STREAM_NAMED("state_tracker", "[REFBOX] Keeping state: " << state2string( m_state.state) );
147  }
