629  Task generateImpl( const string& taskName){
630 
631  auto& def = mTasks[ taskName ];
632  auto& params = def.parameters;
633  const auto& normalTables = def.normalTableTypes;
634  const auto& turnTables = def.ttTypes;
635  const auto& shelfs = def.shTypes;
636  const auto& precisionTable = def.ppTypes;
637  Object::reset();
638  vector<Object> objects;
639 
640  auto availableObjects = extractObjectTypes( taskName );
641  vector<ObjectType> availableCavities = mAvailableCavities; // TODO: filter according to allowed Objects from Task
642  ROS_DEBUG_STREAM_NAMED("generator", "[REFBOX-GEN] Tables:\n" << mTables);
643  ROS_DEBUG_STREAM_NAMED("generator", "[REFBOX-GEN] Cavities:\n" << mAvailableCavities);
644  ROS_DEBUG_STREAM_NAMED("generator", "[REFBOX-GEN] ObjectTypes:\n" << availableObjects);
645 
646  sanityCheck(taskName, availableObjects);
647  auto seedIt = params.find("seed");
648  if ( seedIt != params.end() ) mRand.seed(seedIt->second);
649 
650 
651  // PP fill
652  // generate places on PP
653  // Container generate and distribute
654  // generate places on Container
655  // generate places on shelf
656  // generate remaining places
657  // generate picks on shelf
658  // generate picks on rt
659  // generate remaining picks
660 
661 
662  size_t objectCount = accumulate(availableCavities.begin(), availableCavities.end(), 0,
663  [](size_t n, const ObjectType& t){ return n + t.count; } );
664  objectCount += accumulate(availableObjects.begin(), availableObjects.end(), 0,
665  [](size_t n, const ObjectType& t){ return n + t.count; } );
666 
667  objects.resize(objectCount);
668  auto start = objects.begin();
669 
670  auto cavities = generateCavities( def, availableCavities, start);
671  start += cavities.size();
672  cleanTypes( availableCavities );
673  if ( cavities.size() < params[ "pp_placing" ] ) {
674  ostringstream os;
675  os << "Not enough cavities generated! Generated " << cavities.size() << ", min Necessary: " << params[ "pp_placing" ] << "!";
676  throw runtime_error(os.str());
677  }
678  auto end = cavities.end();
679  for ( size_t i = 0; i < params[ "pp_placing" ]; i++ ) {
680  Object* selected = uniqueSelect<Object*>(cavities.begin(), end);
681  auto filter = [selected](const ObjectType* t){ return t->form != selected->form; };
682  auto genObjects = generateObjects(def, availableObjects, start++, 1, 0, filter);
683  if ( genObjects.empty() ) {
684  throw runtime_error("Did not generate any objects!");
685  }
686  place( def, *genObjects.front(), *selected );
687  }
688  cleanTypes( availableObjects );
689 
690  auto containers = generateContainers( def, availableObjects, start );
691  start += containers.size();
692  cleanTypes( availableObjects );
693  if ( containers.size() < 1 && params[ "container_placing" ] ) {
694  ostringstream os;
695  os << "Not enough containers generated! Generated " << containers.size() << ", min Necessary: 1!";
696  throw runtime_error(os.str());
697  }
698  auto rand = uniform_int_distribution<size_t>( 0, containers.size()-1 );
699  auto genObjects = generateObjects(def, availableObjects, start, params[ "container_placing" ] );
700  start += params[ "container_placing" ];
701  for ( Object* objPtr: genObjects)
702  place( def, *objPtr, *containers[ rand(mRand) ] );
703 
704  size_t created = accumulate(objects.begin(), objects.end(), 0,
705  [](size_t n, const Object& o){ return n + (o.type==Type::OBJECT || o.type == Type::COLORED_OBJECT ? 1 : 0); } );
706  genObjects = generateObjects(def, availableObjects, start, params[ "objects" ] - created, params[ "decoys"] );
707  cleanTypes( availableObjects );
708  start += params[ "objects" ] - created + params[ "decoys" ];
709 
710  size_t shelfes_placing = params[ "shelfes_placing" ];
711  size_t rt_placing = params[ "tt_placing" ];
712  size_t shelfes_picking = params[ "shelfes_picking" ];
713  size_t rt_picking = params[ "tt_picking" ];
714 
715  for ( Object* objPtr: genObjects ) {
716  if ( shelfes_placing ) {
717  place( def, *objPtr, shelfs );
718  shelfes_placing--;
719  continue;
720  }
721  if ( rt_placing ) {
722  place( def, *objPtr, turnTables );
723  rt_placing--;
724  continue;
725  }
726  place(def, *objPtr, normalTables );
727  }
728 
729  for ( Object* objPtr: genObjects ) {
730  if ( shelfes_picking ) {
731  pick( def, *objPtr, shelfs );
732  shelfes_picking--;
733  continue;
734  }
735  if ( rt_picking ) {
736  pick( def, *objPtr, turnTables );
737  rt_picking--;
738  continue;
739  }
740  pick(def, *objPtr, normalTables );
741  }
742 
743  for (auto it = objects.begin(); it < start; it++)
744  if ( !it->source )
745  pick(def, *it, normalTables );
746 
747  objects.erase(start, objects.end());
748 
749  ROS_DEBUG_STREAM_NAMED("generator", "[REFBOX-GEN] Objects:\n" << objects);
750  ROS_DEBUG_STREAM_NAMED("generator", "[REFBOX-GEN] Available Objects:\n" << availableObjects);
751  ROS_DEBUG_STREAM_NAMED("generator", "[REFBOX-GEN] Available Cavities:\n" << availableCavities);
752 
753  return toTask(objects);
754  }
